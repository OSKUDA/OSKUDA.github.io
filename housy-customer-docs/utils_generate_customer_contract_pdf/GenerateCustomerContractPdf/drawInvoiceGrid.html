void drawInvoiceGrid(
  PdfPage page,
  Size pageSize,
  InvoiceRowData rowData,
) {
  var grid = PdfGrid()
    ..style = PdfGridStyle(
      font: invoiceGridFont,
      cellPadding: PdfPaddings(left: 5, right: 5, top: 5, bottom: 5),
    );
  // add 6 columns
  grid.columns.add(count: 8);
  // add 1 header
  grid.headers.add(1);

  var header = grid.headers[0];
  header.cells[0].value = 'No.';
  header.cells[1].value = 'Professional';
  header.cells[2].value = 'Service';
  header.cells[3].value = 'Description';
  header.cells[4].value = 'Qty';
  header.cells[5].value = 'Unit Price';
  header.cells[6].value = 'Tax';
  header.cells[7].value = 'Amount';

  header
    ..style = PdfGridRowStyle(
      backgroundBrush: PdfSolidBrush(PdfColor(23, 68, 214, 255)),
      textBrush: PdfSolidBrush(PdfColor(255, 255, 255, 255)),
      font: pdfFont13,
    )
    ..height = 45;

  // set the width of columns
  grid.columns[0].width = 40;
  grid.columns[1].width = 90;
  grid.columns[2].width = 80;
  // grid.columns[3].width = 120;
  grid.columns[4].width = 40;
  grid.columns[5].width = 50;
  grid.columns[6].width = 50;
  grid.columns[7].width = 60;

  // first row
  var row1 = grid.rows.add();
  row1.cells[0].value = rowData.no;
  row1.cells[1].value = rowData.proName;
  row1.cells[2].value = rowData.serviceName;
  row1.cells[3].value = rowData.description;
  row1.cells[4].value = rowData.numberOfRecurringJobs;
  row1.cells[5].value = rowData.unitPrice;
  row1.cells[6].value = rowData.tax;
  row1.cells[7].value = rowData.amount;

  // set the row height
  row1.height = pageSize.height / 2.0;

  // set row text format
  var rowStringFormat = PdfStringFormat()
    ..alignment = PdfTextAlignment.center
    ..lineAlignment = PdfVerticalAlignment.top;
  // row 1
  row1.cells[0].stringFormat = rowStringFormat;
  row1.cells[4].stringFormat = rowStringFormat;
  row1.cells[5].stringFormat = rowStringFormat;
  row1.cells[6].stringFormat = rowStringFormat;
  row1.cells[7].stringFormat = rowStringFormat;

  // set header text format
  var headerStringFormat = PdfStringFormat()
    ..alignment = PdfTextAlignment.center
    ..lineAlignment = PdfVerticalAlignment.middle;
  header.cells[0].stringFormat = headerStringFormat;
  header.cells[1].stringFormat = headerStringFormat;
  header.cells[2].stringFormat = headerStringFormat;
  header.cells[3].stringFormat = headerStringFormat;
  header.cells[4].stringFormat = headerStringFormat;
  header.cells[5].stringFormat = headerStringFormat;
  header.cells[6].stringFormat = headerStringFormat;
  header.cells[7].stringFormat = headerStringFormat;
  grid.draw(
    page: page,
    bounds: Rect.fromLTWH(0, 220, pageSize.width, 0),
  );
}
