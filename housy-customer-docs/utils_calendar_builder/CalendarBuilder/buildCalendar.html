CalendarBuilders buildCalendar(ActiveJobAttendanceState state) {
  /// [DateTime] of job start date
  var startDate = DateTime.fromMillisecondsSinceEpoch(
      (state.activeJob.date![0] as Timestamp).millisecondsSinceEpoch);

  /// [DateTime] of job end date
  var endDate = DateTime.fromMillisecondsSinceEpoch(
      (state.activeJob.date![1] as Timestamp).millisecondsSinceEpoch);

  /// List of [String] of attended days marked true on attendance data
  var attendedDays = state.attendedDays as List<String>;

  /// List of [String] of absent days marked false on attendance data
  var absentDays = state.absentDays as List<String>;
  return CalendarBuilders(
    defaultBuilder:
        (BuildContext context, DateTime day, DateTime focusedDay) {
      var dayString = DateFormat(DateFormat.YEAR_MONTH_DAY).format(day);
      var formatedDay =
          DateTime.parse(DateFormat('yyyy-MM-dd HH:mm:ss').format(day));
      return attendedDays.contains(dayString)
          ? Container(
              decoration: const BoxDecoration(
                shape: BoxShape.circle,
              ),
              child: Center(
                child: Text(day.day.toString(),
                    style: const TextStyle(
                      color: Color(0xFF16C151),
                      fontWeight: FontWeight.bold,
                    )),
              ),
            )
          : absentDays.contains(dayString)
              ? Container(
                  margin: const EdgeInsets.all(6.0),
                  decoration: const BoxDecoration(
                    shape: BoxShape.circle,
                    color: Color(0xFFFF516E),
                  ),
                  child: Center(
                    child: Text(
                      day.day.toString(),
                      style: const TextStyle(
                        color: Color(0xFFFFFFFF),
                        fontWeight: FontWeight.bold,
                      ),
                    ),
                  ),
                )
              : ((formatedDay.isAfter(startDate) &&
                          formatedDay.isBefore(endDate)) ||
                      (formatedDay == startDate || formatedDay == endDate))
                  ? Container(
                      decoration: const BoxDecoration(
                        shape: BoxShape.circle,
                      ),
                      child: Center(
                        child: Text(
                          day.day.toString(),
                          style: const TextStyle(
                            color: Color(0xFF1744D6),
                            fontWeight: FontWeight.bold,
                          ),
                        ),
                      ),
                    )
                  : Container(
                      decoration: const BoxDecoration(
                        shape: BoxShape.circle,
                      ),
                      child: Center(
                        child: Text(
                          day.day.toString(),
                          style: const TextStyle(
                            color: Color(0xFFC4C4C4),
                            fontWeight: FontWeight.w700,
                          ),
                        ),
                      ),
                    );
    },
  );
}
