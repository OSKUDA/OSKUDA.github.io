void logUserError(String route, {String? reason}) {
  if (!kDebugMode) {
    mixpanel?.identify(FirebaseAuth.instance.currentUser?.uid ?? '');
    mixpanel?.getPeople().increment('user_error', 1);
    mixpanel?.track(
      'User Error',
      properties: {
        'route': route,
        'reason': reason,
      },
    );
  }
}
