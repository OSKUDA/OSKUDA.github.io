void getAddress(JobpostCubit _jobPostCubit, ctx) async {
  var hasSavedAddresses = await DatabaseService().hasSavedAddresses();
  log(hasSavedAddresses.toString());
  if (hasSavedAddresses) {
    await showModalBottomSheet(
      isScrollControlled: true,
      shape: const RoundedRectangleBorder(
        borderRadius: BorderRadius.only(
          topRight: Radius.circular(12),
          topLeft: Radius.circular(12),
        ),
      ),
      context: ctx,
      builder: (_) {
        return BlocProvider.value(
          value: _jobPostCubit,
          child: FutureBuilder<AddressType?>(
              future: DatabaseService().getSelectedAddressType(),
              builder: (context, selectAddressType) {
                if (!selectAddressType.hasData) {
                  return SizedBox(
                    height: 0.2.sh,
                    child: const Center(
                      child: CircularProgressIndicator(),
                    ),
                  );
                }
                return FutureBuilder<List>(
                  future: DatabaseService().getSavedAddressFS(),
                  builder: (context, addressState) {
                    if (!addressState.hasData) {
                      return SizedBox(
                        height: 0.2.sh,
                        child: const Center(
                          child: CircularProgressIndicator(),
                        ),
                      );
                    }
                    return SelectAddressSheet(
                      addressState: addressState.data,
                      selectAddressType: selectAddressType.data,
                    );
                  },
                );
              }),
        );
      },
    );
  }

  if (!hasSavedAddresses) {
    materialNavigateTo(
      child: MultiBlocProvider(
        providers: [
          BlocProvider.value(value: _jobPostCubit),
          BlocProvider(
            create: (_) => AddressCubit()..initLocation(needCSC: true),
          ),
        ],
        child: ConfirmLocation(),
      ),
      context: ctx,
    );
  }
}
