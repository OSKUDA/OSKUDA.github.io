Future<bool> createPdf() async {
  var document = PdfDocument();
  document.pageSettings.margins = PdfMargins()..all = pdfPageMargin;
  final page = document.pages.add();
  final pageSize = page.getClientSize();

  // get indian currency support font
  var hindFontData = await getFontData(GoogleFonts.hind());
  currencySymbolFont = hindFontData.isEmpty
      ? PdfStandardFont(PdfFontFamily.timesRoman, 12)
      : PdfTrueTypeFont(hindFontData, 12);
  // get raleway font
  var ralewayBoldFontData =
      await getFontData(GoogleFonts.raleway(fontWeight: FontWeight.bold));
  var ralewayMediumFontData =
      await getFontData(GoogleFonts.raleway(fontWeight: FontWeight.w400));
  // init the fonts
  initFonts(
    boldFontData: ralewayBoldFontData,
    mediumFontData: ralewayMediumFontData,
  );
  // process row data
  var rowData = getRowData(pro, job);
  var serviceConnectionChargeRowData = InvoiceRowData(
    no: '2',
    proName: '',
    serviceName: '',
    description: 'Service Connection Charges',
    numberOfRecurringJobs: '-',
    unitPrice: '75',
    tax: '0.0',
    amount: '75',
  );
  // generate invoice number
  var invoiceNumber = generateInvoiceNumber(pro, job);

  // invoice date
  var startDate = job.date == null
      ? null
      : job.date![0] == null
          ? null
          : job.date![0]!.millisecondsSinceEpoch;
  var jobEndingDate = job.date == null
      ? null
      : job.date![1] == null
          ? null
          : job.date![1]!.millisecondsSinceEpoch;
  var invoiceDate = DateFormat('MMMM d, yyyy').format(
    DateTime.fromMillisecondsSinceEpoch(
      jobEndingDate ?? Timestamp.now().millisecondsSinceEpoch,
    ),
  );
  var jobStartDate = DateFormat('MMMM d, yyyy').format(
    DateTime.fromMillisecondsSinceEpoch(
      startDate ?? Timestamp.now().millisecondsSinceEpoch,
    ),
  );
  var jobDuration = daysBetween(
      DateTime.fromMillisecondsSinceEpoch(
          startDate ?? Timestamp.now().millisecondsSinceEpoch),
      DateTime.fromMillisecondsSinceEpoch(
          jobEndingDate ?? Timestamp.now().millisecondsSinceEpoch));

  // budget of the job
  var budget = job.proOffers?.where((element) {
    return element['proId'] == pro.pro_uid;
  }).toList()[0]['budget'];

  // due amount
  var dueAmount = (((budget * taxPercentage) + budget) * discountPercentage) +
      ((budget * taxPercentage) + budget);

  // customer details
  var customerName = job.customerName ?? '';
  var customerAddress = job.addressLine ?? '';

  // draw methods
  await drawCompanyDetail(page);
  await drawInvoiceHeader(
      page,
      pageSize,
      invoiceNumber,
      dueAmount.toString(),
      invoiceDate,
      jobStartDate,
      '$jobDuration ${jobDuration > 1 ? 'days' : 'day'}');
  drawBillToShipToDetail(page, pageSize, customerName, customerAddress);
  drawInvoiceGrid(page, pageSize, rowData, serviceConnectionChargeRowData);
  drawPaymentDetails(page, pageSize, budget.toString(), discountPercentage);
  drawRemitTo(page, pageSize);
  drawBottomNote(page, pageSize);
  // clear resources
  var bytes = document.save();
  document.dispose();

  // save file in local storage and open it
  return await saveInStorageAndOpen(invoiceNumber, bytes);
}
