  PdfPage page,

  String amount,
  double discountPercentage,
) {
  var amountWithServiceConnectionCharge =
      (double.parse(amount) + serviceConnectionCharges).toString();
  var discountAmount = (double.parse(amount) * discountPercentage).toString();
  var totalAmount =
      (double.parse(amount) - double.parse(discountAmount)).toString();
  page.graphics.drawString(
    'Amount',
    pdfFont11,
    brush: PdfSolidBrush(
      PdfColor(82, 82, 82),

  );
  page.graphics.drawString(
    amountWithServiceConnectionCharge,
    pdfFont11,
    brush: PdfSolidBrush(
      PdfColor(82, 82, 82),
    ),
    bounds:
        Rect.fromLTRB(pageSize.width / 1.08, bottomSectionTopOffset, 0, 0),
  );
  page.graphics.drawString(
    'Less Service Charge (already paid)',
    pdfFont11,
    brush: PdfSolidBrush(
      PdfColor(82, 82, 82),
    ),
    bounds: Rect.fromLTRB(pageSize.width / 1.75,
        bottomSectionTopOffset + invoiceLineSpacing, 0, 0),
  );
  page.graphics.drawString(
    '- $serviceConnectionCharges',
    pdfFont11,

    bounds: Rect.fromLTRB(pageSize.width / 1.095,

  );
  page.graphics.drawString(
    'Less Discount\r\n(${discountPercentage.toInt()})%',
    pdfFont11,
    brush: PdfSolidBrush(
      PdfColor(82, 82, 82),
    ),
    bounds: Rect.fromLTRB(pageSize.width / 1.75,

  );
  page.graphics.drawString(
    discountAmount,
    pdfFont11,
    brush: PdfSolidBrush(
      PdfColor(82, 82, 82),
    ),
    bounds: Rect.fromLTRB(pageSize.width / 1.08,
        bottomSectionTopOffset + 2.25 * invoiceLineSpacing, 0, 0),
  );

  page.graphics.drawRectangle(
    bounds: Rect.fromLTWH(
        pageSize.width / 1.75,
        bottomSectionTopOffset + 4.5 * invoiceLineSpacing,
        pageSize.width,
        25),
    brush: PdfSolidBrush(PdfColor(23, 68, 214, 255)),
  );

    'Total Amount',
    pdfFont13,
    brush: PdfSolidBrush(
      PdfColor(255, 255, 255),
    ),

        pageSize.width / 1.65,
