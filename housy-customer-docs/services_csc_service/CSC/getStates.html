Future<List<String>> getStates(String country) async {
  var stateList = <String>[];
  try {
    var res = (await Dio().get(
      '${baseUrl}states/$country',
      options: authOption,
    ));

    var source = res.data;

    for (Map state in (source as List)) {
      stateList.add(state['state_name']);
    }
    debugPrint(stateList.toString());
  } catch (e, stack) {
    if (kDebugMode) {
      rethrow;
    }
    log(e.toString(), stackTrace: stack);
  }
  return stateList;
}
