static Route<dynamic> generateRoute(RouteSettings settings) {
  switch (settings.name) {
    case home:
      return MaterialPageRoute(
          settings: settings.copyWith(name: home),
          builder: (_) => const SplashScreen());
    case initProfile:
      return MaterialPageRoute(
          settings: settings.copyWith(name: initProfile),
          builder: (_) => const InitProfile());
    case dashboard:
      return MaterialPageRoute(
        settings: settings.copyWith(name: dashboard),
        builder: (_) => BlocProvider.value(
          value: (settings.arguments as Map)['authCubit'] as AuthCubit,
          child: DashBoard(
            pageIndex: ((settings.arguments as Map)['pageIndex'] ?? 0) as int,
          ),
        ),
      );
    case editProfile:
      return MaterialPageRoute(
          settings: settings.copyWith(name: editProfile),
          builder: (_) => const EditProfile());
    case tawk:
      return MaterialPageRoute(builder: (_) => const TawkScreen());
    case petServices:
      return MaterialPageRoute(
          settings: settings.copyWith(name: petServices),
          builder: (_) => PetServices());
    case allServices:
      return MaterialPageRoute(
          settings: settings.copyWith(name: allServices),
          builder: (_) => const AllServices());
    case applianceRepair:
      return MaterialPageRoute(
          settings: settings.copyWith(name: applianceRepair),
          builder: (_) => const ApplianceRepair());
    case homeCook:
      return MaterialPageRoute(
          settings: settings.copyWith(name: homeCook),
          builder: (_) => const HomeCook());
    case gardening:
      return MaterialPageRoute(
          settings: settings.copyWith(name: gardening),
          builder: (_) => const Gardening());
    case netting:
      return MaterialPageRoute(
          settings: settings.copyWith(name: netting),
          builder: (_) => const Netting());
    case vehicleWash:
      return MaterialPageRoute(
          settings: settings.copyWith(name: vehicleWash),
          builder: (_) => const VehicleWash());
    case tailor:
      return MaterialPageRoute(
          settings: settings.copyWith(name: tailor),
          builder: (_) => const Tailor());
    case mason:
      return MaterialPageRoute(
          settings: settings.copyWith(name: mason),
          builder: (_) => const Mason());
    case sanitationServices:
      return MaterialPageRoute(
          settings: settings.copyWith(name: sanitationServices),
          builder: (_) => const SanitationServices());
    case painter:
      return MaterialPageRoute(
          settings: settings.copyWith(name: painter),
          builder: (_) => const Painter());
    case plumber:
      return MaterialPageRoute(
          settings: settings.copyWith(name: plumber),
          builder: (_) => const Plumber());
    case laundry:
      return MaterialPageRoute(
          settings: settings.copyWith(name: laundry),
          builder: (_) => const Laundry());
    case electrician:
      return MaterialPageRoute(
          settings: settings.copyWith(name: electrician),
          builder: (_) => const Electrician());
    case cobbler:
      return MaterialPageRoute(
          settings: settings.copyWith(name: cobbler),
          builder: (_) => const Cobbler());
    case beauty:
      return MaterialPageRoute(
          settings: settings.copyWith(name: beauty),
          builder: (_) => const Beauty());
    case carpenter:
      return MaterialPageRoute(
          settings: settings.copyWith(name: carpenter),
          builder: (_) => const Carpenter());
    case electronicRepair:
      return MaterialPageRoute(
          settings: settings.copyWith(name: electronicRepair),
          builder: (_) => const ElectronicRepair());
    case notificationScreen:
      return MaterialPageRoute(
          settings: settings.copyWith(name: notificationScreen),
          builder: (_) => const NotificationScreen());
    case jobs:
      return MaterialPageRoute(
        settings: settings.copyWith(name: jobs),
        builder: (_) => BlocProvider(
          create: (context) => JobScreenCubit()..getJobPost(),
          child: const JobScreen(),
        ),
      );
    case jobDetail:
      return MaterialPageRoute(
        settings: settings.copyWith(name: jobs),
        builder: (_) => JobDetailsView(job: settings.arguments as Job),
      );

          postId: settings.arguments.toString(),
        ),

      return MaterialPageRoute(
        settings: settings.copyWith(name: isHired),

      );
    case pros:
      return MaterialPageRoute(
        settings: settings.copyWith(name: pros),
        builder: (_) => ProDetailScreen(
          prosDetail: (settings.arguments as Map)['prosDetail'],
          jobId: (settings.arguments as Map)['jobId'],
        ),
      );
    case proDetail:
      return MaterialPageRoute(
        settings: settings.copyWith(name: proDetail),
        builder: (_) => ProDetailScreen(
          prosDetail: (settings.arguments as Map)['prosDetail'],
          jobId: (settings.arguments as Map)['jobId'],
        ),
      );
    case trackJob:
      return MaterialPageRoute(
        settings: settings.copyWith(name: trackJob),
        builder: (_) => BlocProvider<TrackJobCubit>(
          create: (context) => TrackJobCubit(TrackJobState(
            job: (settings.arguments as Map)['job'],
            pro: (settings.arguments as Map)['prosDetail'],
          )),
          child: const TrackScreen(),
        ),
      );
    case fundContract:
      return MaterialPageRoute(
        settings: settings.copyWith(name: fundContract),
        builder: (_) => FundContractScreen(
          job: (settings.arguments as Map)['job'],
          pro: (settings.arguments as Map)['pro'],
        ),
      );
    case viewContract:
      return MaterialPageRoute(
        settings: settings.copyWith(name: viewContract),
        builder: (_) => ContractScreen(
          job: (settings.arguments as Map)['job'],
          prosDetail: (settings.arguments as Map)['prosDetail'],
        ),
      );
    default:
      return MaterialPageRoute(
        settings: settings.copyWith(name: 'CONTACT FLUTTER DEV IMMEDIATELY'),
        builder: (_) => Scaffold(
          body: Center(
            child: Text('No route defined for ${settings.name}'),
          ),
        ),
      );
  }
