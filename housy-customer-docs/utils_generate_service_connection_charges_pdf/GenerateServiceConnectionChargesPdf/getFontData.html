Future<List<int>> getFontData(TextStyle style) async {
  //Get the external storage directory
  var directory = await getApplicationSupportDirectory();
  //Create an empty file to write the font data
  var file = File('${directory.path}/${style.fontFamily}.ttf');
  List<int>? fontBytes;
  //Check if entity with the path exists
  if (file.existsSync()) {
    fontBytes = await file.readAsBytes();
  }
  if (fontBytes != null && fontBytes.isNotEmpty) {
    //Return the fontByte
    return fontBytes;
  } else {
    //Return the empty fontByte
    return [];
  }
}
