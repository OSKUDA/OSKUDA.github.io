void drawInvoiceGrid(
  PdfPage page,
  Size pageSize,
  InvoiceRowData rowData,
) {
  var grid = PdfGrid()
    ..style = PdfGridStyle(
      font: invoiceGridFont,
      cellPadding: PdfPaddings(left: 5, right: 5, top: 5, bottom: 5),
    );
  // add 3 columns
  grid.columns.add(count: 3);
  // add 1 header
  grid.headers.add(1);

  var header = grid.headers[0];
  header.cells[0].value = 'Date';
  header.cells[1].value = 'Description';
  header.cells[2].value = 'Amount';

  header
    ..style = PdfGridRowStyle(
      backgroundBrush: PdfSolidBrush(PdfColor(23, 68, 214, 255)),
      textBrush: PdfSolidBrush(PdfColor(255, 255, 255, 255)),
      font: pdfFont13,
    )
    ..height = 45;

  // set the width of columns
  grid.columns[0].width = 120;
  // grid.columns[1].width = 90;
  grid.columns[2].width = 120;

  // first row
  var row1 = grid.rows.add();
  row1.cells[0].value = rowData.date;
  row1.cells[1].value = 'Service Connection Fee';
  row1.cells[2].value = rowData.amount;

  // set the row height
  row1.height = pageSize.height / 7.5;

  // set row text format
  var rowStringFormat = PdfStringFormat()
    ..alignment = PdfTextAlignment.center
    ..lineAlignment = PdfVerticalAlignment.top;
  // row 1
  row1.cells[0].stringFormat = rowStringFormat;
  row1.cells[2].stringFormat = rowStringFormat;

  // set header text format
  var headerStringFormat = PdfStringFormat()
    ..alignment = PdfTextAlignment.center
    ..lineAlignment = PdfVerticalAlignment.middle;
  header.cells[0].stringFormat = headerStringFormat;
  header.cells[1].stringFormat = headerStringFormat;
  header.cells[2].stringFormat = headerStringFormat;
  grid.draw(
    page: page,
    bounds: Rect.fromLTWH(0, 150, pageSize.width, 0),
  );
}
