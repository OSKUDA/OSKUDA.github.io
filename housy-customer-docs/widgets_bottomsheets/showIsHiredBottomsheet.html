Future showIsHiredBottomsheet({
  required String proId,
  required Job job,
  required Pro proDetails,
  required BuildContext context,
  bool showName = false,
}) async {
  var housyAnalytics = HousyAnalytics();
  return await showModalBottomSheet(
    context: context,
    isScrollControlled: true,
    isDismissible: false,
    // enableDrag: true,
    shape: roundedRectangleBorder.copyWith(
      borderRadius: const BorderRadius.only(
        topLeft: Radius.circular(12),
        topRight: Radius.circular(12),
      ),
    ),
    builder: (_) {
      debugPrint('Bottomsheet opened');
      return Wrap(
        children: [
          Container(
            // height: 240.h,
            padding: EdgeInsets.all(28.sp),
            child: Column(
              crossAxisAlignment: CrossAxisAlignment.start,
              children: [
                if (showName)
                  CachedNetworkImage(
                    imageUrl: proDetails.profile_url ??
                        '''https://cdn3d.iconscout.com/3d/premium/thumb/user-3711850-3105265.png''',
                    imageBuilder: (context, imageProvider) => CircleAvatar(
                      foregroundImage: imageProvider,
                    ),
                  ),
                if (showName) SizedBox(height: 24.h),
                Text(
                  '''Do you want to hire ${showName ? (proDetails.fullName) : '''this Service Provider'''}?''',
                  style: TextStyle(
                    fontSize: 24.sp,
                  ),
                ),
                const Divider(),
                SizedBox(
                  height: 50.h,
                  child: Row(
                    children: [
                      Expanded(
                        child: FilledMaterialButton(
                          color: kPinkAccentColor,
                          text: 'Hire',
                          onPressed: () async {
                            housyAnalytics
                              ..logContractGenerated()
                              ..logGenerateLead(
                                  proDetails.service_type_selected);
                            Navigator.of(context).pop();
                            materialNavigateTo(
                                child: FundContractScreen(
                                    job: job, pro: proDetails),
                                context: context);
                          },
                        ),
                      ),
                      SizedBox(width: 8.0.w),
                      Expanded(
                        child: BorderMaterialButton(
                          color: kPinkAccentColor,
                          text: 'Not Yet',
                          onPressed: () async {
                            Navigator.of(context).pop();
                          },
                        ),
                      ),
                    ],
                  ),
                ),
                SizedBox(width: 8.0.w),
                Center(
                  child: BorderMaterialButton(
                    color: kPinkAccentColor,
                    text: 'Reject this professional',
                    onPressed: () async {
                      await showDialog(
                          context: context,
                          builder: (_) {
                            return Dialog(
                              child: Padding(
                                padding: EdgeInsets.all(24.sp),
                                child: Wrap(
                                  children: [
                                    Text(
                                      '''Are you sure you want to reject ?''',
                                      style: bold18TextStyle,
                                    ),
                                    Text(
                                      '''If rejected this professional will be removed from the offers section.''',
                                      style: normal14TextStyle,
                                    ),
                                    SizedBox(height: 32.h),
                                    Align(
                                      alignment: Alignment.centerRight,
                                      child: MaterialButton(
                                        onPressed: () async {
                                          await DatabaseService()
                                              .addRejectedPro(
                                                  proId, job.documentID!);

                                          housyAnalytics.logProRejected();
                                          Navigator.of(context).pop();
                                          Navigator.of(context).pop();
                                        },
                                        child: const Text(
                                          '''Yes. I'm Sure''',
                                          style: blueTextStyle,
                                        ),
                                      ),
                                    ),
                                  ],
                                ),
                              ),
                            );
                          });
                    },
                  ),
                ),
              ],
            ),
          ),
        ],
      );
    },
  );
}
