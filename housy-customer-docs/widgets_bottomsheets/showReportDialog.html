                              blurRadius: 1,

                            ),
                          ],
                        ),
                        child: Align(
                          alignment: Alignment.centerLeft,
                          child: Text(
                            'Report Issues',
                            style: bold18TextStyle.copyWith(
                              fontSize: 24.0,
                            ),
                          ),
                        ),
                      ),
                      BlocBuilder<ReportDialogCubit, ReportDialogState>(
                        builder: (context, state) {

                                    ),

                                          borderSide: const BorderSide(
                                              color: Colors.white),

                                              BorderRadius.circular(12.0),

                                          context
                                              .read<ReportDialogCubit>()
                                              .reportDetailSubmitted(
                                                textEditingController.text,
                                                prosDetail,
                                                FirebaseAuth
                                                    .instance.currentUser!.uid,
                                              );
                                        },
                                        color: kPrimaryColor,
                                      ),
                                    ),

                                ),

                                        onPressed: () {},

                                          shape: const CircleBorder(),

                                        child: state.submitSuccess
                                            ? const Icon(
                                                Icons.check,
                                                color: Color(0xFFFFFFFF),

                                                Icons.clear,

                                              CrossAxisAlignment.stretch,

                                            Text(
                                                state.submitSuccess
