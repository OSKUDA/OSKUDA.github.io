Future<void> setupInteractedMessage(BuildContext context) async {
  // ! Get any messages which caused the application to open from
  // !  a terminated state.

  var initialMessage = await FirebaseMessaging.instance.getInitialMessage();

  if (initialMessage != null) {
    log(initialMessage.data.toString());
    if (initialMessage.data['route'] == '/offer') {
      dynamic noti;
      if (initialMessage.notification != null) {
        if (initialMessage.notification!.title!.contains('New')) {
          noti = Noti(
            id: initialMessage.messageId ?? initialMessage.sentTime.toString(),
            notiType: NotiTypes.newOffer,
            title: initialMessage.notification!.title ?? 'New Offer Alert',
            body: initialMessage.notification!.body ??
                'You have new offer on our job posting',
            timestamp:
                Timestamp.fromDate(initialMessage.sentTime ?? DateTime.now()),
            extra: initialMessage.data,
          );
        } else {
          noti = Noti(
            id: initialMessage.messageId ?? initialMessage.sentTime.toString(),
            notiType: NotiTypes.jobUpdate,
            title: initialMessage.notification!.title ?? 'Job Status Update',
            body: initialMessage.notification!.body ??
                'Professinal has updated the job status',
            timestamp:
                Timestamp.fromDate(initialMessage.sentTime ?? DateTime.now()),
            extra: initialMessage.data,
          );
        }
        if (!context.read<NotificationstorageCubit>().checkIfExists(noti)) {
          context.read<NotificationstorageCubit>().addToNotiStorage(noti);
        }
      }
    }

    if (initialMessage.data['route'] == '/review') {
      String proId = initialMessage.data['id'];
      var jobID = initialMessage.data['postid'] ?? '';
      var proDetails = await DatabaseService().getProDetail(proId);
      var jobDetails = await DatabaseService().getJobPost(jobID);
      var noti = Noti(
        id: initialMessage.messageId ?? initialMessage.sentTime.toString(),
        notiType: NotiTypes.isHired,
        title: initialMessage.notification!.title ??
            '''Did you hire ${proDetails.fullName} yet?''',
        body: initialMessage.notification!.body ??
            '''Click here to rate them, tell us what you liked or disliked''',
        timestamp:
            Timestamp.fromDate(initialMessage.sentTime ?? DateTime.now()),
        extra: initialMessage.data,
      );

      if (!context.read<NotificationstorageCubit>().checkIfExists(noti)) {
        context.read<NotificationstorageCubit>().addToNotiStorage(noti);
        debugPrint(
            '\n\n\nNav Triggered Closed!!!!------------------------------------------------------------------------->\n\n\n');
        // ignore: unawaited_futures
        Navigator.pushNamed(
          context,
          '/trackJob',
          arguments: {
            'prosDetail': proDetails,
