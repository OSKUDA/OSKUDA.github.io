List<List<String>> getSuggestions(String query) {
  var suggestion = List.of(dialCodeAndCountryList).where((data) {
    final dataLower = data[1].toLowerCase();
    final queryLower = query.toLowerCase();
    return dataLower.contains(queryLower);
  }).toList();
  if (suggestion.first[1].contains(query) &&
      (suggestion.length == 1 || suggestion.length == 2)) {
    var auxilaryList = dialCodeAndCountryList;
    for (var element in suggestion) {
      auxilaryList.remove(element);
    }
    suggestion = List.from(suggestion)..addAll(auxilaryList);
  }
  return suggestion;
}
