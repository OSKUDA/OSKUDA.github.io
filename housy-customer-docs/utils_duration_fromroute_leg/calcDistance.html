String calcDistance(List<Leg> distances) {
  var value = 0;
  var suffix = 'm';
  try {
    distances.forEach((element) => value += (element.distance?.value as int?)!);
    if (value > 999) {
      // meters to kilometers
      value = (value / 1000).round();
      suffix = 'km';
    }
  } catch (e) {
    log(e.toString());
  }
  return '$value $suffix';
}
