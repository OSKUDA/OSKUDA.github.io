String calcDuration(List<Leg> durations) {
  var value = 0;
  var suffix = 'sec';
  try {
    durations.forEach(
        (element) => value += (element.durationInTraffic?.value as int?)!);
    if (value > 60) {
      // seconds to minutes
      value = (value / 60).round();
      suffix = value == 1 ? 'min' : 'mins';
    }
    if (value > 60) {
      // minutes to hours
      value = (value / 60).round();
      suffix = 'hr';
    }
    if (value > 24) {
      // hours to days
      value = (value / 24).round();
      suffix = 'd';
    }
  } catch (e) {
    log(e.toString());
  }
  return '$value $suffix';
}
