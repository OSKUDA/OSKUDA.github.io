void getActiveJob(List<Job> jobList, String proUid, Job? job) {
  var myActiveJobList = <Job>[];
  if (job == null) {
    jobList.forEach((job) {
      job.hiredPros?.forEach((hiredProUid) {
        if (hiredProUid == proUid &&
            (job.jobStatus!.status == 'Active' ||
                job.jobStatus!.status == 'Accepted')) {
          myActiveJobList.add(job);
        }
      });
    });
  } else {
    myActiveJobList.add(job);
  }

  if (myActiveJobList.isNotEmpty) {
    var activeJob = myActiveJobList[0];
    var jobTitle = activeJob.jobCategory!;
    var isSingleJob = activeJob.jobType == 'One-Time' ? true : false;
    final startDateTime = (activeJob.date![0] as Timestamp).toDate();
    final startDate = DateFormat('d MMM').format(startDateTime);
    final budget = activeJob.proOffers?.where((element) {
      return element['proId'] == proUid;
    }).toList()[0]['budget'];
    var workDays = daysBetween(
      (activeJob.date![0] as Timestamp).toDate(),
      (activeJob.date![1] as Timestamp).toDate(),
    );
    final timeslots = activeJob.proOffers?.where((element) {
          return element['proId'] == proUid;
        }).toList()[0]['timeSlot'] ??
        [];
    // ? Here, timeslots are received as string. Checked jobStatus .fromMap
    // ? .toMap .toJson methods. Couldn't find an issue.
    // ? Implemented manual type check below to solve the issue.
    dynamic startTime;
    dynamic endTime;
    if (timeslots[0]['start'] is String) {
      startTime = Timestamp.fromDate(DateTime.parse(timeslots[0]['start']));
      endTime = Timestamp.fromDate(DateTime.parse(timeslots[0]['end']));
    } else {
      // ? expected timstamp type
      startTime = timeslots[0]['start'];
      endTime = timeslots[0]['end'];
    }
    final calendarStartTime =
        '''${(startTime as Timestamp).toDate().hour}:${(startTime).toDate().minute < 10 ? (startTime).toDate().minute.toString().padLeft(2, '0') : (startTime).toDate().minute}''';
    final calendarEndTime =
        '''${(endTime).toDate().hour}:${(endTime).toDate().minute < 10 ? (endTime).toDate().minute.toString().padLeft(2, '0') : (endTime).toDate().minute}''';

    // calendar view variables
    var focusedDay = startDateTime;
    var selectedDay = DateTime.now();
    var calendarFormat = CalendarFormat.week;
    var attendance = activeJob.jobStatus!.attendance;
    emit(
      ActiveJobAttendanceLoaded(
        activeJob: activeJob,
        jobTitle: jobTitle,
        workDays: workDays,
        timeslots: timeslots,
        attendedDays: const <String>[],
        absentDays: const <String>[],
        attendance: attendance ?? {},
        budget: budget.toString(),
        isSingleJob: isSingleJob,
        startDate: startDate,
        startDateTime: startDateTime,
        startTime: calendarStartTime,
        endTime: calendarEndTime,
        focusedDay: focusedDay,
        selectedDay: selectedDay,
        calendarFormat: calendarFormat,
      ),
    );
    // should be called after emiting ActiveJobAttendanceLoaded state
    // following method depends on state data of above emited state
    updateAttendance();
  }
}
