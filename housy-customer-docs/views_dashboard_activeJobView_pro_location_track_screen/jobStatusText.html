String jobStatusText(String status) {
  switch (status) {
    case 'Open':
      return 'Arriving at location';
    case 'Left':
      return 'Arriving at location';
    case 'Reached':
      return 'Reached at location';
    case 'On-Job':
      return 'Service In Progress';
    case 'Awaiting-OTP':
      return 'OTP Verification';
    case 'Finished':
      return 'Job Completed';
    default:
      return 'Arriving at location';
  }
}
