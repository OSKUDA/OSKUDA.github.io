void slidePushTo({
  required Widget child,
  required BuildContext context,
  Object? args,
  required Widget currentChild,
}) {
  Navigator.push(
    context,
    PageTransition(
      type: PageTransitionType.rightToLeftWithFade,
      child: child,
      childCurrent: const AllServices(),
      settings: RouteSettings(arguments: args),
    ),
  );
}
