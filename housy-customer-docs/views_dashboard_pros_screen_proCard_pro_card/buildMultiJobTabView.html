List<Widget> buildMultiJobTabView({
  required List<Job> jobs,
  required String proId,
  required Pro prosDetail,
}) {
  try {
    var multiJobTabViewList = <Widget>[];

    jobs.forEach((job) {
      var _budget = job.proOffers!.where((proOfferMap) {
        if (proOfferMap['proId'] == proId) {
          return true;
        }
        return false;
      }).first['budget'];

      var _isActive = job.jobStatus?.status == 'Ongoing' ? true : false;
      var _date =
          DateFormat('d MMM').format((job.date?[0] as Timestamp).toDate());

      final time = job.proOffers?.where((element) {
            return element['proId'] == prosDetail.pro_uid;
          }).toList()[0]['timeSlot']?[0] ??
          {
            'start': Timestamp.now(),
            'end': Timestamp.now(),
          };

      final timeStart = time?['start'] is Timestamp
          ? (time['start'] as Timestamp).toDate()
          : DateTime.parse(time['start']);

      final _startTime =
          '''${timeStart.hour > 12 ? timeStart.hour - 12 : timeStart.hour}:${timeStart.minute.toString().length == 1 ? timeStart.minute.toString().padRight(2, '0') : timeStart.minute}''';
      final _isStartTimeInPM = timeStart.hour / 12 > 1 ? true : false;

      multiJobTabViewList.add(MultiJobTabView(
        budget: _budget.toString(),
        startDate: _date,
        startTime: _startTime,
        isStartTimeInPm: _isStartTimeInPM,
        isActiveJob: _isActive,
        endDate:
            DateFormat('d MMM').format((job.date?[1] as Timestamp).toDate()),
        job: job,
      ));
    });

    return multiJobTabViewList;
  } catch (e, s) {
    log(e.toString(), stackTrace: s);
    return [];
  }
}
