Future<void> signIn(String contact) async {
  try {
    await _auth.verifyPhoneNumber(
      phoneNumber: contact,
      codeAutoRetrievalTimeout: (String verId) {
        verificationId = verId;
      },
      codeSent: (String verId, [int? forceCodeResend]) {
        verificationId = verId;
        debugPrint('***********$verificationId *************');
        debugPrint('OTP Sent');
        Fluttertoast.showToast(msg: 'OTP Sent');
      },
      timeout: const Duration(seconds: 120),
      verificationCompleted: (PhoneAuthCredential credential) async {
        debugPrint(
            '****************PHone Creden : $credential ****************');
        await FirebaseAuth.instance.signInWithCredential(credential).then(
          (value) async {
            if (value.user != null) {
              var user = value.user!;
              var isNewUser = value.additionalUserInfo!.isNewUser ||
                  !(await DatabaseService().searchUser(user));
              if (isNewUser) {
                housyAnalytics.logSignUp();

                emit(
                  Authenticated(
                    username: user.displayName ?? '',
                    uid: user.uid,
                    isNewUser: isNewUser,
                  ),
                );
                onlineStatusSocket.online(user.uid);
              } else {
                var userData = await DatabaseService().getUser();
                housyAnalytics.logSignIn();
                emit(
                  Authenticated(
                    email: userData.email,
                    location: userData.location,
                    mobile: userData.mobile,
                    nearestAddressLine: userData.nearestAddressLine,
                    notifyTokens: userData.notifyTokens,
                    activeJobs: userData.activeJobs,
                    snapshot: userData.snapshot,
                    reference: userData.reference,
                    username: userData.fullName ?? user.displayName ?? '',
                    uid: user.uid,
                    gender: userData.gender ?? user.displayName ?? '',
                    isNewUser: false,
                  ),
                );
                onlineStatusSocket.online(user.uid);
              }

              await showToast(
                'OTP Verification Successful!',
                Colors.green.shade300,
              );
            }
          },
        );
      },
      verificationFailed: (FirebaseAuthException exception) {
        debugPrint(exception.message);
        handleAuthError(exception);
        emit(UnAuthenticated());
      },
    );
  } catch (e) {
    handleAuthError(e as FirebaseAuthException);
    emit(UnAuthenticated());
  }
}
