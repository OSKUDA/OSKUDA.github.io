Future<void> getPros() async {
  /// List of [ProScreenMap] of hired professionals
  var hiredProList = <ProScreenMap>[];

  /// List of [ProScreenMap] of contacted professionals
  var contactedProList = <ProScreenMap>[];

  /// List of [ProScreenMap] of past professionals
  var pastProList = <ProScreenMap>[];

  /// List of pro unique id of hired professionals
  var hiredProIdList = <Map<String, dynamic>>[];

  /// List of pro unique id of contacted professionals
  var contactedProIdList = <Map<String, dynamic>>[];

  /// List of pro unique id of past professionals
  var pastProIdList = <Map<String, dynamic>>[];

  /// Adds the [Map] instance to [hiredProIdList]
  ///
  /// Emit [ProScreenLoaded] with empty [hiredProIdList] if hired pro is empty
  /// Else, populates the [hiredProIDList]
  void addToHiredProIdList(Job job) {
    if ((job.hiredPros ?? []).isEmpty ||
        job.jobStatus!.status == 'Completed') {
      emit(ProScreenLoaded(
        const [],
        state.contactedProList,
        state.pastProList,
      ));
    } else {
      // hiredProIdList.addAll({...?job.hiredPros,job});
      hiredProIdList.addAll(
        [
          {
            'proId': job.hiredPros,
            'job': job,
          }
        ],
      );
    }
  }

  /// Adds the [Map] instance to [contactedProIdList]
  ///
  /// Emit [ProScreenLoaded] with empty [contactedProIdList] if contacted pro
  /// is empty
  /// Else, populates the [contactedProIdList]
  void addToContactedProIdList(Job job) {
    if ((job.contactedPros ?? []).isEmpty) {
      emit(ProScreenLoaded(
        state.hiredProList,
        const [],
        state.pastProList,
      ));
    } else {
      contactedProIdList.addAll(
        [
          {
            'proId': job.contactedPros,
            'job': job,
          }
        ],
      );
    }
  }

  /// Adds the [Map] instance to [pastProIdList]
  ///
  /// Emit [ProScreenLoaded] with empty [pastProIdList] if past pro is empty
  /// Else, populates the [pastProIdList]
  void addToPastProIdList(Job job) {
    // todo
    if ((job.pastPros ?? []).isEmpty &&
        job.jobStatus!.status != 'Completed') {
      emit(ProScreenLoaded(
        state.hiredProList,
        state.contactedProList,
        const [],
      ));
    } else {
      pastProIdList.addAll(
        [
          {
            'proId': job.hiredPros,
            'job': job,
          }
        ],
      );
    }
  }

  /// Populates the [hiredProList] according to [listOfHiredProId]
  ///
  /// If the pro id exists in hiredProList, append the job to the existing
  /// [ProScreenMap]
  /// Else, add the new [ProScreenMap] instance to hiredProList
  ///
  /// Emit [ProScreenLoaded] with empty [hiredProList] if hire pro is empty
  /// Else, emit [ProScreenLoaded] with [hiredProList] data
  void addToHiredProList(Map map) async {
    log('****************REACHEDHIRED**************');

    var listOfHiredProId = map['proId'] as List<String?>?;
    var job = map['job'] as Job;

    await Future.forEach(listOfHiredProId!, (hiredProId) async {
      var exist = false;
      var _proDetail =
          await _databaseService.getProDetail(hiredProId.toString());

      for (var index = 0; index < hiredProList.length; index++) {
        if (hiredProList[index].proId == hiredProId) {
          hiredProList[index].jobs.add(job);
          exist = true;
        }
      }

      if (exist == false) {
        hiredProList.addAll([
          ProScreenMap(
            proId: hiredProId.toString(),
            proDetail: _proDetail,
            jobs: [job],
          )
        ]);
      }
    });

    if (hiredProList.isEmpty) {
      emit(ProScreenLoaded(
        const [],
        state.contactedProList,
        state.pastProList,
      ));
    } else {
      emit(ProScreenLoaded(
        hiredProList,
        state.contactedProList,
        state.pastProList,
      ));
    }
  }

  /// Populates the [contactedProList] according to [listOfContactedproId]
  ///
  /// If the pro id exists in contactedProList, append the job to the existing
  /// [ProScreenMap]
  /// Else, add the new [ProScreenMap] instance to contactedProList
  ///
  /// Emit [ProScreenLoaded] with empty [contactedProList] if contacted pro is
  /// empty
  /// Else, emit [ProScreenLoaded] with [contactedProList] data
  void addToContactedProList(Map map) async {
    // var pro = await _databaseService.getProDetail(proId!);
    log('****************RECHEDCONTACTED**************');

    var listOfContactedproId = map['proId'] as List<String?>?;
    var job = map['job'] as Job;

    await Future.forEach(listOfContactedproId!, (contactedProId) async {
      var exist = false;
      var _proDetail =
          await _databaseService.getProDetail(contactedProId.toString());

      for (var index = 0; index < contactedProList.length; index++) {
        if (contactedProList[index].proId == contactedProId) {
          contactedProList[index].jobs.add(job);
          exist = true;
        }
      }

      if (exist == false) {
        contactedProList.addAll([
          ProScreenMap(
            proId: contactedProId.toString(),
            proDetail: _proDetail,
            jobs: [job],
          )
        ]);
      }
    });

    if (contactedProList.isEmpty) {
      // emit(ContactedProScreenEmpty(state.hiredProList));
      emit(ProScreenLoaded(
        state.hiredProList,
        const [],
        state.pastProList,
      ));
    } else {
      emit(ProScreenLoaded(
        state.hiredProList,
        contactedProList,
        state.pastProList,
      ));
    }
  }

  /// Populates the [PastProList] according to [listOfPastproId]
  ///
  /// If the pro id exists in PastProList, append the job to the existing
  /// [ProScreenMap]
  /// Else, add the new [ProScreenMap] instance to PastProList
  ///
  /// Emit [ProScreenLoaded] with empty [PastProList] if past pro is empty
  /// Else, emit [ProScreenLoaded] with [PastProList] data
  void addToPastProList(Map map) async {
    log('****************RECHEDCONTACTED**************');

    var listOfPastproId = map['proId'] as List<String?>?;
    var job = map['job'] as Job;

    await Future.forEach(listOfPastproId!, (pastProId) async {
      var exist = false;
      var _proDetail =
          await _databaseService.getProDetail(pastProId.toString());

      for (var index = 0; index < pastProList.length; index++) {
        if (pastProList[index].proId == pastProId) {
          pastProList[index].jobs.add(job);
          exist = true;
        }
      }

      if (exist == false) {
        pastProList.addAll([
          ProScreenMap(
            proId: pastProId.toString(),
            proDetail: _proDetail,
            jobs: [job],
          )
        ]);
      }
    });

    if (pastProList.isEmpty) {
      emit(ProScreenLoaded(
        state.hiredProList,
        state.contactedProList,
        const [],
      ));
    } else {
      emit(ProScreenLoaded(
        state.hiredProList,
        state.contactedProList,
        pastProList,
      ));
    }
  }

  /// Subscribe and listen to the stream subscription on [getJobPosts]
  ///
  /// Calls methods [addToHiredProList], [addToContactedProList], and
  /// [addToPastProList] to update the [hiredProList], [contactedProList], and
  /// [pastProIdList]
  streamSub = (await _databaseService.getJobPosts()).listen((querySnapshot) {
    debugPrint(' No of jobs found = ${querySnapshot.size.toString()}');

    emit(const ProScreenLoading());

    if (querySnapshot.size == 0) {
      emit(const ProScreenEmpty());
      return;
    }

    var _jobList = <Job>[];
    querySnapshot.docs.forEach(
      (QueryDocumentSnapshot element) {
        try {
          _jobList.add(
              Job.fromFirestore(element).copyWith(documentID: element.id));
        } catch (e, stack) {
          if (kDebugMode) {
            rethrow;
          }
          log(e.toString(), stackTrace: stack);
        }
      },
    );
    if (_jobList.isNotEmpty) {
      hiredProList = <ProScreenMap>[];
      contactedProList = <ProScreenMap>[];
      pastProList = <ProScreenMap>[];
      hiredProIdList = <Map<String, dynamic>>[];
      contactedProIdList = <Map<String, dynamic>>[];
      pastProIdList = <Map<String, dynamic>>[];
      for (var job in _jobList) {
        addToContactedProIdList(job);
        addToHiredProIdList(job);
        addToPastProIdList(job);
      }

      hiredProIdList.forEach(addToHiredProList);
      contactedProIdList.forEach(addToContactedProList);
      pastProIdList.forEach(addToPastProList);
    }
  });
}
