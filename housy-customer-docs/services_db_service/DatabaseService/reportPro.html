/// Adds report posted by customer on [Pro]'s document
/// 'numberOfReports' => +1
///
/// 'true' if successfully updated the report
/// 'false' if it fails
Future<bool> reportPro(
  String proUid,
  String customerUid,
  String reportType,
  String reportDetail,
) async {
  var result = false;
  await firestoreInstance?.collection('professionals').doc(proUid).update({
    'numberOfReports': FieldValue.increment(1),
    'reports': FieldValue.arrayUnion(
      [
        {
          'reportType': reportType,
          'reportDetail': reportDetail,
          'reportedBy': customerUid,
        },
      ],
    ),
  }).onError((dynamic error, stackTrace) {
    // throw error;
    debugPrint(error);
  }).then((value) {
    result = true;
  });
  return result;
}
