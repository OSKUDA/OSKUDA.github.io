@override
Map<String, dynamic>? toJson(List<Noti> state) {
  Map<String, dynamic>? map = {};

  void addNotiToMap(Noti noti) {
    map.putIfAbsent(noti.id, () => noti.toJson());
  }

  // debugPrint(map.toString() + '++++++++++++++');

  try {
    state.forEach(addNotiToMap);
  } catch (e, stackTrace) {
    debugPrint('${e.toString()}\n' '${stackTrace.toString()}');
    FirebaseCrashlytics.instance.recordError(e, stackTrace);
  }

  return map;
}
